## 一、设计目标

### 1.1 核心原则

1. **策略无关** - 框架不绑定任何特定策略，打板只是策略的一种
2. **模块化** - 各模块独立，可单独测试和替换
3. **可扩展** - 新策略/数据源/风控规则可插拔
4. **高性能** - 向量化引擎，支持大规模数据回测
5. **实盘对接** - 回测 → 每日选股 → 实盘信号链路

### 1.2 支持策略类型

| 策略类型 | 描述 | 实现优先级 |
|----------|------|------------|
| 打板策略 | 涨停板追涨，23维评分 | P0 |
| 缩量潜伏策略 | 涨停后缩量回调低吸 | P0 |
| 板块轮动策略 | 热点板块轮动 | P1 |
| 多因子策略 | 因子选股+组合优化 | P1 |
| 均值回归策略 | 超跌反弹 | P2 |
| 趋势跟踪策略 | 动量突破 | P2 |

#### 3.1.2 策略注册器

```python
class StrategyRegistry:
    """策略注册器 - 支持动态加载策略"""
    
    _strategies: Dict[str, type] = {}
    
    @classmethod
    def register(cls, strategy_class: type) -> type:
        """注册策略"""
        instance = strategy_class()
        cls._strategies[instance.strategy_name] = strategy_class
        return strategy_class
    
    @classmethod
    def get(cls, name: str) -> type:
        """获取策略类"""
        if name not in cls._strategies:
            raise ValueError(f"策略 '{name}' 未注册")
        return cls._strategies[name]
    
    @classmethod
    def list_all(cls) -> List[str]:
        """列出所有已注册策略"""
        return list(cls._strategies.keys())
    
    @classmethod
    def from_config(cls, config_path: str) -> 'BaseStrategy':
        """从配置文件加载策略"""
        import yaml
        with open(config_path) as f:
            config = yaml.safe_load(f)
        
        strategy_name = config.get('strategy_name')
        strategy_class = cls.get(strategy_name)
        return strategy_class(config)
```

```python
class DataCache:
    """数据缓存管理"""
    
    def __init__(self, cache_dir: str = "/data/agents/master/cache"):
        self.cache_dir = cache_dir
        self._memory_cache: Dict[str, pd.DataFrame] = {}
    
    def get(self, key: str) -> Optional[pd.DataFrame]:
        """获取缓存"""
        # 先查内存缓存
        if key in self._memory_cache:
            return self._memory_cache[key]
        
        # 再查磁盘缓存
        cache_file = f"{self.cache_dir}/{key}.parquet"
        if os.path.exists(cache_file):
            df = pd.read_parquet(cache_file)
            self._memory_cache[key] = df
            return df
        
        return None
    
    def set(self, key: str, data: pd.DataFrame, ttl: int = None):
        """设置缓存"""
        self._memory_cache[key] = data
        cache_file = f"{self.cache_dir}/{key}.parquet"
        data.to_parquet(cache_file)
    
    def invalidate(self, key: str):
        """失效缓存"""
        if key in self._memory_cache:
            del self._memory_cache[key]
        cache_file = f"{self.cache_dir}/{key}.parquet"
        if os.path.exists(cache_file):
            os.remove(cache_file)
```

```python
        # 策略无关的回测逻辑

            # 1. 获取数据

            limit_data = data_loader.get_limit_list(date)
```

```python
class PositionManager:
    """仓位管理器"""
    
    def __init__(self, config: Dict):
        self.max_positions = config.get('max_positions', 2)
        self.position_ratio = config.get('position_ratio', 0.2)
        self.mode = config.get('position_mode', 'fixed')  # fixed/volatility/kelly
    
    def calculate_position(self, 
                          signal: Signal, 
                          capital: float,
                          volatility: float = None) -> int:
        """计算建议仓位"""
        if self.mode == 'fixed':
            return int(capital * self.position_ratio / signal.price / 100) * 100
        elif self.mode == 'volatility':
            # 波动率调整仓位
            target_risk = 0.02  # 目标风险2%
            pos_value = capital * target_risk / volatility
            return int(pos_value / signal.price / 100) * 100
        elif self.mode == 'kelly':
            # 凯利公式
            win_rate = signal.confidence
            win_loss_ratio = signal.metadata.get('win_loss_ratio', 1.5)
            kelly = win_rate - (1 - win_rate) / win_loss_ratio
            return int(capital * kelly * 0.5 / signal.price / 100) * 100  # 半凯利
    
    def can_open(self, signal: Signal) -> bool:
        """检查是否可以开仓"""
        if len(self.positions) >= self.max_positions:
            return False
        # 其他约束检查
        return True
```

```python
class StopLossManager:
    """止损管理器"""
    
    def __init__(self, config: Dict):
        self.stop_loss = config.get('stop_loss', 0.06)
        self.stop_profit = config.get('stop_profit', 0.12)
        self.max_hold_days = config.get('max_hold_days', 3)
        self.use_trailing = config.get('use_trailing_stop', False)
    
    def check(self, positions: Dict[str, Position], date: str) -> List[Signal]:
        """检查止损止盈"""
        signals = []
        for code, pos in positions.items():
            # 价格止损
            if pos.profit_loss_pct <= -self.stop_loss:
                signals.append(Signal(
                    ts_code=code,
                    signal_type=SignalType.SELL,
                    reason='stop_loss'
                ))
            # 价格止盈
            elif pos.profit_loss_pct >= self.stop_profit:
                signals.append(Signal(
                    ts_code=code,
                    signal_type=SignalType.SELL,
                    reason='stop_profit'
                ))
            # 时间止损
            elif pos.hold_days >= self.max_hold_days:
                signals.append(Signal(
                    ts_code=code,
                    signal_type=SignalType.SELL,
                    reason='time_stop'
                ))
        return signals
```

```python
class PerformanceCalculator:

    def calculate(self,

        """计算完整绩效指标"""
        
        values = [d['total'] for d in daily_capital]
        returns = pd.Series([d['return'] for d in daily_capital])
        
        # 收益指标
        total_return = (values[-1] - values[0]) / values[0]
        annual_return = (1 + total_return) ** (252 / len(values)) - 1
        
        # 风险指标
        volatility = returns.std() * np.sqrt(252)
        max_drawdown = self._calc_max_drawdown(values)
        
        # 风险调整指标
        sharpe_ratio = returns.mean() / returns.std() * np.sqrt(252) if returns.std() > 0 else 0
        sortino_ratio = self._calc_sortino(returns)
        calmar_ratio = annual_return / max_drawdown if max_drawdown > 0 else 0
        
        # 交易指标
        win_trades = [t for t in trades if t.get('pnl', 0) > 0]
        loss_trades = [t for t in trades if t.get('pnl', 0) < 0]
        win_rate = len(win_trades) / len(trades) if trades else 0
        profit_loss_ratio = np.mean([t['pnl'] for t in win_trades]) / abs(np.mean([t['pnl'] for t in loss_trades])) if loss_trades else 0
        
        # Alpha/Beta (需要基准)
        alpha, beta = 0, 1
        if benchmark_returns is not None:
            alpha, beta = self._calc_alpha_beta(returns, benchmark_returns)
        
        return {
            'return': {
                'total_return': total_return,
                'annual_return': annual_return,
                'excess_return': annual_return - 0.03  # 假设无风险利率3%
            },
            'risk': {
                'volatility': volatility,
                'max_drawdown': max_drawdown,
                'var_95': np.percentile(returns, 5),
                'var_99': np.percentile(returns, 1)
            },
            'risk_adjusted': {
                'sharpe_ratio': sharpe_ratio,
                'sortino_ratio': sortino_ratio,
                'calmar_ratio': calmar_ratio,
                'alpha': alpha,
                'beta': beta,
                'information_ratio': self._calc_information_ratio(returns, benchmark_returns)
            },
            'trade': {
                'total_trades': len(trades),
                'win_rate': win_rate,
                'profit_loss_ratio': profit_loss_ratio,
                'avg_hold_days': np.mean([t.get('hold_days', 0) for t in trades]),
                'turnover_rate': self._calc_turnover(trades, values)
            }
        }
```

```python
class DailyPicker:
    """每日选股器"""
    
    def __init__(self, strategy: BaseStrategy, data_source: DataSource):
        self.strategy = strategy
        self.data_source = data_source
    
    def pick(self, date: str = None) -> List[Signal]:
        """
        执行每日选股
        
        Args:
            date: 选股日期，默认为最新交易日
            
        Returns:
            选股信号列表
        """
        if date is None:
            date = self._get_latest_trade_date()
        
        # 获取数据
        daily_data = self.data_source.get_daily(...)
        limit_data = self.data_source.get_limit_list(date)
        
        # 策略筛选
        filtered = self.strategy.filter(daily_data, date)
        scored = self.strategy.score(filtered, date)
        
        # 生成信号
        signals = []
        for _, row in scored.head(5).iterrows():
            signals.append(Signal(
                ts_code=row['ts_code'],
                signal_type=SignalType.BUY,
                price=row.get('close', 0),
                shares=0,  # 由仓位管理器计算
                confidence=row.get('total_score', 0) / 100,
                reason=f"策略选股: {self.strategy.strategy_name}",
                timestamp=date
            ))
        
        return signals
```

```python
class Notifier:
    """通知服务"""
    
    def __init__(self, config: Dict):
        self.wechat_url = config.get('wechat_webhook_url')
        self.feishu_url = config.get('feishu_webhook_url')
    
    def send_signal(self, signals: List[Signal], date: str):
        """发送选股信号"""
        message = self._format_message(signals, date)
        
        # 企业微信
        if self.wechat_url:
            self._send_wechat(message)
        
        # 飞书
        if self.feishu_url:
            self._send_feishu(message)
    
    def _format_message(self, signals: List[Signal], date: str) -> str:
        """格式化消息"""
        lines = [f"📊 每日选股信号 ({date})"]
        for i, sig in enumerate(signals, 1):
            lines.append(f"{i}. {sig.ts_code} | 置信度: {sig.confidence:.2f} | {sig.reason}")
        return "\n".join(lines)
```

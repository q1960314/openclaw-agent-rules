### P1-2: 移动止损

**设计**:
```python
class TrailingStopLoss:
    """移动止损"""
    
    def __init__(self, activation_pct=0.05, trail_pct=0.03):
        self.activation_pct = activation_pct  # 激活阈值
        self.trail_pct = trail_pct  # 回撤阈值
        self.high_prices = {}  # 最高价追踪
    
    def update(self, ts_code, current_price):
        if ts_code not in self.high_prices:
            self.high_prices[ts_code] = current_price
        else:
            self.high_prices[ts_code] = max(self.high_prices[ts_code], current_price)
    
    def should_stop(self, ts_code, current_price, cost_price):
        # 盈利超过激活阈值后，回撤超过trail_pct则止损
        profit_pct = (current_price - cost_price) / cost_price
        if profit_pct < self.activation_pct:
            return False
        
        high = self.high_prices.get(ts_code, current_price)
        drawdown = (high - current_price) / high
        return drawdown > self.trail_pct
```

**涉及文件**: `risk_engine.py` 或新建 `stop_loss.py`

### P2-1: 实盘信号通知

**设计**:
```python
class SignalNotifier:
    """实盘信号通知器"""
    
    def __init__(self, wechat_url=None, feishu_url=None):
        self.wechat_url = wechat_url
        self.feishu_url = feishu_url
    
    def send_daily_picks(self, signals, date):
        """发送每日选股信号"""
        message = self._format_message(signals, date)
        if self.feishu_url:
            self._send_feishu(message)
        if self.wechat_url:
            self._send_wechat(message)
    
    def _format_message(self, signals, date):
        lines = [f"📊 每日选股 ({date})"]
        for i, sig in enumerate(signals, 1):
            lines.append(f"{i}. {sig.ts_code} | 评分:{sig.score:.1f} | 置信:{sig.confidence:.2f}")
        return "\n".join(lines)
```

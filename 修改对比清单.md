### 4. 数据压缩优化 - 新增压缩方法

**位置：** Utils 类中

**修改后新增方法：**
```python
def save_to_parquet_snappy(self, df, file_path, data_type="日线行情"):
    """
    【优化】保存数据到 Parquet 文件，使用 Snappy 压缩
    :param df: DataFrame
    :param file_path: 保存路径
    :param data_type: 数据类型（用于日志）
    """
    if df.empty:
        return
    
    if not PARQUET_AVAILABLE:
        # 降级为 CSV
        df.to_csv(file_path.replace('.parquet', '.csv'), index=False, encoding='utf-8-sig')
        logger.warning(f"⚠️  pyarrow 不可用，{data_type}降级为 CSV 存储：{file_path.replace('.parquet', '.csv')}")
        return
    
    try:
        # 计算压缩比日志
        original_size = df.memory_usage(deep=True).sum()
        
        # 保存为 Parquet + Snappy 压缩
        table = pa.Table.from_pandas(df)
        pq.write_table(table, file_path, compression='snappy')
        
        # 计算压缩后大小
        compressed_size = os.path.getsize(file_path)
        compression_ratio = original_size / compressed_size if compressed_size > 0 else 0
        
        logger.info(f"✅ {data_type}已保存（Parquet+Snappy）：{file_path}")
        logger.info(f"📊 压缩比日志 | {data_type}: 原始{original_size/1024/1024:.2f}MB → 压缩后{compressed_size/1024/1024:.2f}MB | 压缩比：{compression_ratio:.2f}x")
    except Exception as e:
        logger.warning(f"⚠️  {data_type}Parquet 保存失败：{e}，降级为 CSV")
        df.to_csv(file_path.replace('.parquet', '.csv'), index=False, encoding='utf-8-sig')

def save_to_sqlite(self, df, db_path, table_name, data_type="财务数据"):
    """
    【优化】保存数据到 SQLite 数据库
    :param df: DataFrame
    :param db_path: 数据库文件路径
    :param table_name: 表名
    :param data_type: 数据类型（用于日志）
    """
    if df.empty:
        return
    
    if not SQLITE_AVAILABLE:
        logger.warning(f"⚠️  sqlite3 不可用，{data_type}无法存储")
        return
    
    try:
        conn = sqlite3.connect(db_path)
        df.to_sql(table_name, conn, if_exists='replace', index=False)
        conn.close()
        
        # 计算数据库文件大小
        db_size = os.path.getsize(db_path)
        
        logger.info(f"✅ {data_type}已保存（SQLite）：{db_path} (表：{table_name})")
        logger.info(f"📊 压缩比日志 | {data_type}: 数据库大小{db_size/1024/1024:.2f}MB")
    except Exception as e:
        logger.warning(f"⚠️  {data_type}SQLite 保存失败：{e}")
```

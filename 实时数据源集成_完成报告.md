```python
def get_realtime_data(stock_code: str):
    # 优先级：Tushare → AkShare → 新浪 → 东财
    try:
        return tushare_realtime(stock_code)
    except:
        try:
            return akshare_realtime(stock_code)
        except:
            try:
                return sina_crawler(stock_code)
            except:
                return eastmoney_crawler(stock_code)
```

## 五、使用示例

### 5.1 最简单用法

```python
from modules.data_sources import get_realtime_data

# 获取单只股票（自动选择最佳数据源）
data = get_realtime_data("000001")
if data:
    print(f"{data['ts_code']}: ¥{data['price']} ({data['pct_change']}%)")
    print(f"数据源：{data['source']}")
```

### 5.2 批量获取

```python
from modules.data_sources import get_realtime_batch

# 批量获取股票行情
batch = get_realtime_batch(["000001", "600519", "300750"])
for code, data in batch.items():
    print(f"{code}: {data['price']}")
```

### 5.3 获取板块数据

```python
from modules.data_sources import get_industry_board_realtime

# 获取行业板块实时行情
df = get_industry_board_realtime()
print(df.nlargest(5, 'pct_change')[['board_name', 'pct_change']])
```

### 5.4 获取资金流向

```python
from modules.data_sources import get_moneyflow

# 获取单只股票资金流向
flow = get_moneyflow("000001")
print(f"主力净流入：{flow['main_net_inflow']:.2f}万元")
```
